body::after {
	content             : " ";
	display             : inline-block;
	/*** 丸 ***/
	width               : 要素のwidth;
	height              : 要素のheight;
	border-radius       : widthの半分;
	/*** 背景：画像を適用したいためcontentではなくbackgroundを利用 ***/

	background-size     : contain;
	background-position : center center;
	/*** 下部にセンタリング ***/
	position            : absolute;
	left                : 0;
	right               : 0;
	bottom              : 0;
	margin              : auto;
	/*** OnsenUIの前面に表示するため ***/
	z-index             : 999;
}
